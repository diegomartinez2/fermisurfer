41 /**
42  @brief Compute coefficient for the French-curve (Kumo) interpolation
43  @f[
44  A^{\rm intp} = \sum_{i = 1}^4 C_i A_i^{\rm orig}
45  @f]
46 */
47 static void kumo_coef(
48  int j, //!< [in] Interpolated grid index
49  GLfloat *coef, //!< [out] Coefficient of interpolation @f$C_i@f$
50  int interpol
51 ) {
52  GLfloat x, mx;
53  x = (GLfloat)j / (GLfloat)interpol;
54  mx = 1.0f - x;
55  coef[0] = -0.5f * x * mx * mx;
56  coef[1] = mx * (mx*mx + 3.0f* x*mx + 0.5f* x* x);
57  coef[2] = x * ( x* x + 3.0f*mx* x + 0.5f*mx*mx);
58  coef[3] = -0.5f * x * x * mx;
59 }
